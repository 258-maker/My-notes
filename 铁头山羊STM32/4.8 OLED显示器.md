# OLED显示器的基本原理
> OLED显示器内部有一个内存，可以通过I2C总线对这个内存进行读写，内存里的每一个比特位都代表屏幕上的一个像素，向对0应的比特位写0就点亮，写1就熄灭

# 屏幕初始化
## 初始化软I2C
```c
#include "si2c.h"//引用软件I2C的头文件 si2c。h
SI2C_TypeDef si2c;//声明一个软i2c的变量

si2c.SCL_GOIOx=GPIOB;//指定SCL的位置
si2c.SCL_GOIO_Pin=GPIO_Pin_6;
si2c.SDA_GOIOx=GPIOB;//指定SDA的位置
si2c.SDA_GOIO_Pin=GPIO_Pin_7;
SI2C_Init(&si2c);//初始化
```
```c
#include "stm32f10x.h"
#include "si2c.h"
SI2C_TypeDef si2c;
void My_SoftwareI2C_Init(void);
int main(void)
{

	while(1)
	{
	}
```

## 屏幕初始化代码编写
```c
int OLED_Init(OLED_TypeDef *OLED,    // 所使用的OLED的名称
              OLED_InitTypeDef *OLED_InitStruct);  // OLED的初始化参数
```
作用：对OLED显示器进行初始化
```c
struct OLED_InitTypeDef{
    // i2c写数据回调函数
    int (*i2c_write_cb)(uint8_t addr, const uint8_t *pdata, uint16_t size);
}
```

==1. 按照回调函数的形式写一个函数==
==2. 把函数名填进去==

```c
#include "stm32f10x.h"
#include "si2c.h"
#include "oled.h"
SI2C_TypeDef si2c;
OLED_TypeDef oled;
void My_SoftwareI2C_Init(void);
void My_OLEDScreen_Init(void);
int i2c_write_bytes(uint8_t addr,const uint8_t *pdata,uint16_t size);
int main(void)
{
	My_SoftwareI2C_Init();//对软I2C进行初始化
	My_OLEDScreen_Init();//对OLED显示器进行初始化
	while(1)
	{
	}
}

void My_SoftwareI2C_Init(void)
{
	si2c.SCL_GPIOx=GPIOB;
	si2c.SCL_GPIO_Pin=GPIO_Pin_6;
	si2c.SDA_GPIOx=GPIOB;
	si2c.SDA_GPIO_Pin=GPIO_Pin_7;
	My_SI2C_Init(&si2c);
}

int i2c_write_bytes(uint8_t addr,const uint8_t *pdata,uint16_t size)
{
	return My_SI2C_SendBytes(&si2c,addr,pdata,size);
}

void My_OLEDScreen_Init(void)
{
	OLED_InitTypeDef OLED_InitStruct;
	OLED_InitStruct.i2c_write_cb=i2c_write_bytes;
	OLED_Init(&oled,&OLED_InitStruct);
}

```

# 文字相关的操作
## 打印字符串
```c
int OLED_DrawString(OLED_TypeDef *OLED,  // 所使用的OLED的名称
                    const char *Str);    // 要显示的字符串

OLED_SetBrush(&oled, BRUSH_TRANSPARENT); // 透明画刷
OLED_SetPen(&oled, PEN_COLOR_WHITE, 1);  // 白色画笔
OLED_SetCursor(&oled, 24,50);            // 光标(25,50)
OLED_DrawString(&oled, "Hello world");   // 打印Hello world
```
**具体例子**
```c
#include "stm32f10x.h"
#include "si2c.h"
#include "oled.h"
SI2C_TypeDef si2c;
OLED_TypeDef oled;
void My_SoftwareI2C_Init(void);
void My_OLEDScreen_Init(void);
int i2c_write_bytes(uint8_t addr,const uint8_t *pdata,uint16_t size);
int main(void)
{
	My_SoftwareI2C_Init();//对软I2C进行初始化
	My_OLEDScreen_Init();//对OLED显示器进行初始化
	
	//#1 打印Hello world
	OLED_SetPen(&oled,PEN_COLOR_WHITE,4);
	OLED_SetBrush(&oled,PEN_COLOR_TRANSPARENT);
	OLED_SetCursor(&oled,24,50);
	OLED_DrawString(&oled, "Hello world");
	
	OLED_SendBuffer(&oled);
	while(1)
	{
	}
}

	My_SI2C_Init(&si2c);
}

int i2c_write_bytes(uint8_t addr,const uint8_t *pdata,uint16_t size)
{
	return My_SI2C_SendBytes(&si2c,addr,pdata,size);
}

void My_OLEDScreen_Init(void)
{
	OLED_InitTypeDef OLED_InitStruct;
```
