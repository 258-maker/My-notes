# 初始化IO引脚
- PA0引脚模式：模拟模式（输入模拟信号）
```c
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA,ENABLE);
	GPIO_InitTypeDef GPIO_InitStruct={0};
	GPIO_InitStruct.GPIO_Mode=GPIO_Mode_AIN;
	GPIO_InitStruct.GPIO_Pin=GPIO_Pin_0;
	GPIO_Init(GPIOA,&GPIO_InitStruct);
```
# 配置ADC的时钟
示例代码
```c
//设置分频器的分频系数（6分频）
RCC_ADCCLKConfig(RCC_PCLK2_Div6);
//使能ADC1的时钟
RCC_APB2PeriphClockCmd(RCC_APB2Periph_ADC1,ENABLE);
```

# 初始化ADC的基本常数
![](../图片/IMG_20260219_195921.jpg)
- 左对齐与右对齐
- 一般使用右对齐
![](../图片/IMG_20260219_200312.jpg)
```c
	ADC_InitTypeDef ADC_InitStruct={0};
	ADC_InitStruct.ADC_ContinuousConvMode=DISABLE;//关闭连续模式
	ADC_InitStruct.ADC_DataAlign=ADC_DataAlign_Right;//右对齐
	ADC_InitStruct.ADC_ExternalTrigConv=ADC_ExternalTrigConv_None;//软件启动
	ADC_InitStruct.ADC_Mode=ADC_Mode_Independent;//独立通道
	ADC_InitStruct.ADC_NbrOfChannel=1;//常规序列1个通道
	ADC_InitStruct.ADC_ScanConvMode=DISABLE;
	
	ADC_Init(ADC1,&ADC_InitStruct);
```
# 配置常规序列
- 常规序列：通道0，10.24Cycle
```c
	ADC_RegularChannelConfig(ADC1,ADC_Channel_0,1,ADC_SampleTime_13Cycles5);//采样周期10.24Cycle
	ADC_ExternalTrigConvCmd(ADC1,ENABLE);
```

# 启动并读取转换结果
![](../图片/IMG_20260219_203405.jpg)
```c
		//#1 清除EOC标志位
		ADC_ClearFlag(ADC1,ADC_FLAG_EOC);
		
		//#2 通过软件启动的方式发送脉冲
		ADC_SoftwareStartConvCmd(ADC1,ENABLE);
		
		//#3 等待常规序列转换完成
		while(ADC_GetFlagStatus(ADC1,ADC_FLAG_EOC)==RESET);
			
		//#4 读取结果
		uint16_t dr=ADC_GetConversionValue(ADC1);
		
		//#5把结果转换为电压
		float voltage=dr*(3.3f/4095);
```
